Install the full Karabiner Elements, and carefully follow all of the prompts
within Karabiner to make changes in the MacOS System Settings, which included,
if I remember correctly, the following:

#+begin_src text :tangle "no"
  Login Items & Extensions > Karabiner-Elements Non-Privileged Agents (enable)
  Login Items & Extensions > Karabiner-Elements Privileged Daemons (enable)
  Login Items & Extensions > Driver Extensions > .Karabiner-VirtualHIDDevice-Manager (enable)
#+end_src

And not sure if all of these are required, but somewhere along the way I enabled these:
Privacy & Security > Input Monitoring, enable for:

#+begin_src text :tangle "no"
  kanata
  Karabiner-Core-Service
  kitty
#+end_src

Then, importantly, QUIT Karabiner the app, AND, importantly, find and quit the
Karabiner running in the MacOS menu bar (at the top of the screen). IF you
don't do this, then Kanata will complain that something else is using the
keyboard, and won't work.

Possibly / Probably reboot again.

THEN try running Kanata like so: ~sudo kanata --cfg your-config.kbd~

#+begin_src kanata-kbd :tangle (if (eq system-type 'darwin) "/sudo::/opt/homebrew/etc/kanata.kbd" "no")
  (defcfg

  (defsrc

  ;; https://precondition.github.io/home-row-mods#gasc

    lA (tap-hold $tap-time $hold-time l ralt)
    ;G (tap-hold $tap-time $hold-time ; rmet)

  (deflayer base
    caps brdn brup _ _ _ _ prev pp next mute vold volu
    XX
    esc @aG @sA @dS @fC @jC @kS @lA @;G XX
    XX XX
    @fnl XX XX bspc ret XX

    caps f1 f2 f3 f4 f5 f6 f7 f8 f9 f10 f11 f12
    _

    _ _

      local path=$(eval echo "${3//>}") # santize input so find doesn't break on spaces or ~
      local -i fileperms="$2" dirperms="$1"
      find "$path" -type d -exec chmod "$dirperms" {} \;
      find "$path" -type f -exec chmod "$fileperms" {} \;

  memsum() {
      ps -eo size,pid,user,command --sort -size | \
          awk '{ hr=$1/1024 ; printf("%13.2f Mb ",hr) } { for ( x=4 ; x<=NF ; x++ ) { printf("%s ",$x) } print "" }' | \
          cut -d "" -f2 | \
          cut -d "-" -f1| \
          grep "${1}" | grep -v grep
  }

  memusage() {
      #Report Total Used and Available mem in human readable format
      total=$(head -1 /proc/meminfo |awk '{print $2}')
      avail=$(head -2 /proc/meminfo |tail -1 |awk '{print $2}')
      used=$(( total - avail ))
      totalMB=$(( total / 1024 ))
      availMB=$(( avail / 1024 ))
      usedMB=$(( used / 1024 ))
      echo -ne "${totalMB} MB total, ${usedMB} MB used, ${availMB} MB free.\n"
  }

          for p in "${pid[@]}"; do
              if ps -p "$p" &> /dev/null; then
                  echo "Killing $1 process: $p"
                  sudo kill -9 "$p"
              fi
          done
      else
          echo "No $1 processes found."
      fi
  }

  }
